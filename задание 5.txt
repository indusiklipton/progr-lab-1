class Student:
    def __init__(self, last_name, age, group, course):
        self.last_name = last_name
        self.age = age
        self.group = group
        self.course = course

def main():
    students = []
    
    print("Введите данные о студентах (для завершения введите 'стоп' в поле фамилии):")
    
    while True:
        last_name = input("\nФамилия студента: ")
        if last_name.lower() == 'стоп':
            break
            
        age = int(input("Возраст: "))
        group = input("Группа: ")
        course = int(input("Курс: "))
        
        student = Student(last_name, age, group, course)
        students.append(student)
        print("Студент добавлен!")
    
    print("\n" + "="*50)
    print("ПОИСК СТУДЕНТОВ")
    print("="*50)
    
    while True:
        print("\nКритерии поиска:")
        print("1 - По фамилии")
        print("2 - По возрасту")
        print("3 - По группе")
        print("4 - По курсу")
        print("0 - Выход")
        
        choice = input("Выберите критерий поиска: ")
        
        if choice == '0':
            break
        elif choice == '1':
            search_value = input("Введите фамилию: ")
            results = [s for s in students if s.last_name.lower() == search_value.lower()]
        elif choice == '2':
            search_value = int(input("Введите возраст: "))
            results = [s for s in students if s.age == search_value]
        elif choice == '3':
            search_value = input("Введите группу: ")
            results = [s for s in students if s.group.lower() == search_value.lower()]
        elif choice == '4':
            search_value = int(input("Введите курс: "))
            results = [s for s in students if s.course == search_value]
        else:
            print("Неверный выбор!")
            continue
        
        if results:
            print(f"\nНайдено {len(results)} студент(ов):")
            for i, student in enumerate(results, 1):
                print(f"{i}. {student.last_name}, {student.age} лет, {student.group}, {student.course} курс")
        else:
            print("Студенты не найдены!")

if __name__ == "__main__":
    main()
